732

1050  //*************************************************************************
1054  //*************************************************************************
1055  template <typename TCompare>
1056  void sort(TCompare compare)
1057  {
1058  iterator p_left;
1059  iterator p_right;
1060  iterator p_node;
1061  iterator p_head;
1062  iterator p_tail;
1063  int list_size = 1;
1064  int number_of_merges;
1065  int left_size;
1066  int right_size;
1067 
1068  if (is_trivial_list())
1069  {
1070  return;
1071  }
1072 
1073  while (true)
1074  {
1075  p_left = begin();
1076  p_head = before_begin();
1077  p_tail = before_begin();
1078 
1079  number_of_merges = 0; // Count the number of merges we do in this pass.
1080 
1081  while (p_left != end())
1082  {
1083  ++number_of_merges; // There exists a merge to be done.
1084  p_right = p_left;
1085  left_size = 0;
1086 
1087  // Step 'list_size' places along from left
1088  for (int i = 0; i < list_size; ++i)
1089  {
1090  ++left_size;
1091 
1092  ++p_right;
1093 
1094  if (p_right == end())
1095  {
1096  break;
1097  }
1098  }
1099 
1100  // If right hasn't fallen off end, we have two lists to merge.
1101  right_size = list_size;
1102 
1103  // Now we have two lists. Merge them.
1104  while (left_size > 0 || (right_size > 0 && p_right != end()))
1105  {
1106  // Decide whether the next node of merge comes from left or right.
1107  if (left_size == 0)
1108  {
1109  // Left is empty. The node must come from right.
1110  p_node = p_right;
1111  ++p_right;
1112  --right_size;
1113  }
1114  else if (right_size == 0 || p_right == end())
1115  {
1116  // Right is empty. The node must come from left.
1117  p_node = p_left;
1118  ++p_left;
1119  --left_size;
1120  }
1121  else if (!compare(*p_right, *p_left))
1122  {
1123  // First node of left is lower or same. The node must come from left.
1124  p_node = p_left;
1125  ++p_left;
1126  --left_size;
1127  }
1128  else
1129  {
1130  // First node of right is lower. The node must come from right.
1131  p_node = p_right;
1132  ++p_right;
1133  --right_size;
1134  }
1135 
1136  // Add the next node to the merged head.
1137  if (p_head == before_begin())
1138  {
1139  join(p_head.p_node, p_node.p_node);
1140  p_head = p_node;
1141  p_tail = p_node;
1142  }
1143  else
1144  {
1145  join(p_tail.p_node, p_node.p_node);
1146  p_tail = p_node;
1147  }
1148 
1149  p_tail.p_node->next = nullptr;
1150  }
1151 
1152  // Now left has stepped `list_size' places along, and right has too.
1153  p_left = p_right;
1154  }
1155 
1156  // If we have done only one merge, we're finished.
1157  if (number_of_merges <= 1) // Allow for number_of_merges == 0, the empty head case
1158  {
1159  return;
1160  }
1161 
1162  // Otherwise repeat, merging lists twice the size
1163  list_size *= 2;
1164  }
1165  }
