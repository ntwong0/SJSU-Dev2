771  std::pair<iterator, bool> insert(const value_type& key_value_pair)
772  {
773  std::pair<iterator, bool> result(end(), false);
774 
775  ETL_ASSERT(!full(), ETL_ERROR(unordered_map_full));
776 
777  const key_type& key = key_value_pair.first;
778 
779  // Get the hash index.
780  size_t index = get_bucket_index(key);
781 
782  // Get the bucket & bucket iterator.
783  bucket_t* pbucket = pbuckets + index;
784  bucket_t& bucket = *pbucket;
785 
786  // The first one in the bucket?
787  if (bucket.empty())
788  {
789  // Get a new node.
790  node_t& node = *pnodepool->allocate<node_t>();
791  ::new (&node.key_value_pair) value_type(key_value_pair);
792  ETL_INCREMENT_DEBUG_COUNT
793 
794  // Just add the pointer to the bucket;
795  bucket.insert_after(bucket.before_begin(), node);
796 
797  result.first = iterator((pbuckets + number_of_buckets), pbucket, pbucket->begin());
798  result.second = true;
799 
800  adjust_first_last_markers(pbucket);
801  }
802  else
803  {
804  // Step though the bucket looking for a place to insert.
805  local_iterator inode_previous = bucket.before_begin();
806  local_iterator inode = bucket.begin();
807 
808  while (inode != bucket.end())
809  {
810  // Do we already have this key?
811  if (inode->key_value_pair.first == key)
812  {
813  break;
814  }
815 
816  ++inode_previous;
817  ++inode;
818  }
819 
820  // Not already there?
821  if (inode == bucket.end())
822  {
823  // Get a new node.
824  node_t& node = *pnodepool->allocate<node_t>();
825  ::new (&node.key_value_pair) value_type(key_value_pair);
826  ETL_INCREMENT_DEBUG_COUNT
827 
828  // Add the node to the end of the bucket;
829  bucket.insert_after(inode_previous, node);
830  ++inode_previous;
831 
832  result.first = iterator((pbuckets + number_of_buckets), pbucket, inode_previous);
833  result.second = true;
834  }
835  }
