148  template <Port port>
149  static void I2cHandler()
150  {
151  static constexpr uint8_t kPort = util::Value(port);
152  MasterState state = MasterState(i2c[kPort]->STAT);
153  uint32_t clear_mask = 0;
154  uint32_t set_mask = 0;
155  switch (state)
156  {
157  case MasterState::kBusError: // 0x00
158  {
159  transaction[kPort].status = Status::kBusError;
160  set_mask = Control::kAssertAcknowledge | Control::kStop;
161  break;
162  }
163  case MasterState::kStartCondition: // 0x08
164  {
165  i2c[kPort]->DAT = transaction[kPort].address;
166  break;
167  }
168  case MasterState::kRepeatedStart: // 0x10
169  {
170  i2c[kPort]->DAT = transaction[kPort].address;
171  break;
172  }
173  case MasterState::kSlaveAddressWriteSentRecievedAck: // 0x18
174  {
175  clear_mask = Control::kStart;
176  if (transaction[kPort].out_length == 0)
177  {
178  SetBusyState(port, false);
179  transaction[kPort].status = Status::kSuccess;
180  set_mask = Control::kStop;
181  }
182  else
183  {
184  size_t position = transaction[kPort].position++;
185  i2c[kPort]->DAT = transaction[kPort].data_out[position];
186  }
187  break;
188  }
189  case MasterState::kSlaveAddressWriteSentRecievedNack: // 0x20
190  {
191  clear_mask = Control::kStart;
192  SetBusyState(port, false);
193  transaction[kPort].status = Status::kDeviceNotFound;
194  set_mask = Control::kStop;
195  break;
196  }
197  case MasterState::kTransmittedDataRecievedAck: // 0x28
198  {
199  if (transaction[kPort].position >= transaction[kPort].out_length)
200  {
201  if (transaction[kPort].repeated)
202  {
203  // OR with 1 to set address as READ for the next transaction
204  transaction[kPort].address |= 0b1;
205  transaction[kPort].position = 0;
206  set_mask = Control::kStart;
207  }
208  else
209  {
210  SetBusyState(port, false);
211  set_mask = Control::kStop;
212  }
213  }
214  else
215  {
216  size_t position = transaction[kPort].position++;
217  i2c[kPort]->DAT = transaction[kPort].data_out[position];
218  }
219  break;
220  }
221  case MasterState::kTransmittedDataRecievedNack: // 0x30
222  {
223  SetBusyState(port, false);
224  set_mask = Control::kStop;
225  break;
226  }
227  case MasterState::kArbitrationLost: // 0x38
228  {
229  set_mask = Control::kStart;
230  break;
231  }
232  case MasterState::kSlaveAddressReadSentRecievedAck: // 0x40
233  {
234  if (transaction[kPort].in_length == 0)
235  {
236  clear_mask = Control::kAssertAcknowledge | Control::kStart;
237  }
238  else
239  {
240  clear_mask = Control::kStart;
241  set_mask = Control::kAssertAcknowledge;
242  }
243  break;
244  }
245  case MasterState::kSlaveAddressReadSentRecievedNack: // 0x48
246  {
247  clear_mask = Control::kStart;
248  transaction[kPort].status = Status::kDeviceNotFound;
249  SetBusyState(port, false);
250  set_mask = Control::kStop;
251  break;
252  }
253  case MasterState::kRecievedDataRecievedAck: // 0x50
254  {
255  const size_t kBufferEnd = transaction[kPort].in_length;
256  if (transaction[kPort].position < kBufferEnd)
257  {
258  const size_t kPosition = transaction[kPort].position;
259  transaction[kPort].data_in[kPosition] =
260  static_cast<uint8_t>(i2c[kPort]->DAT);
261  transaction[kPort].position++;
262  }
263  // Check if the position has been pushed past the buffer length
264  if (transaction[kPort].position < kBufferEnd)
265  {
266  set_mask = Control::kAssertAcknowledge;
267  }
268  else
269  {
270  SetBusyState(port, false);
271  clear_mask = Control::kAssertAcknowledge;
272  }
273  break;
274  }
275  case MasterState::kRecievedDataRecievedNack: // 0x58
276  {
277  SetBusyState(port, false);
278  if (transaction[kPort].in_length != 0)
279  {
280  size_t position = transaction[kPort].position++;
281  transaction[kPort].data_in[position] =
282  static_cast<uint8_t>(i2c[kPort]->DAT);
283  }
284  set_mask = Control::kStop;
285  break;
286  }
287  case MasterState::kDoNothing: // 0xF8
288  {
289  break;
290  }
291  default:
292  {
293  clear_mask = Control::kStop;
294  SJ2_ASSERT_FATAL(false, "Invalid I2C State Reached!!");
295  break;
296  }
297  }
298  // Clear I2C Interrupt flag
299  clear_mask |= Control::kInterrupt;
300  // Set register controls
301  i2c[kPort]->CONCLR = clear_mask;
302  i2c[kPort]->CONSET = set_mask;
303  }
